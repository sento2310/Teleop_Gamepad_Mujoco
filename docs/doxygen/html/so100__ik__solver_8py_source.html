12class SO100IKSolver:
13 """
14 @brief 5DOF IK solver that preserves manually controlled joints
15 @details Implements position-only inverse kinematics while fixing
16 rotation, wrist_pitch, and wrist_roll joints to prevent
17 conflicts between manual control and IK solutions.
18 """
19
20 def __init__(self, model, data, end_effector_body, joint_controller):
21 """
22 @brief Initialize IK solver for SO100 arm
23
24 @param model: MuJoCo model object containing robot kinematics
25 @param data: MuJoCo data object for current simulation state
26 @param end_effector_body: Name of end effector body for IK targeting
27 @param joint_controller: SO100Teleoperation instance for joint mapping
28
29 @throws ValueError: If end_effector_body not found in model
30
31 @note Builds joint mapping and defines control strategy automatically
32 """
33 self.model, self.data = model, data
34 self.joint_controller = joint_controller
35
36 # Get end-effector body ID
37 self.ee_body_id = mujoco.mj_name2id(model, mujoco.mjtObj.mjOBJ_BODY, end_effector_body)
38 if self.ee_body_id == -1:
39 raise ValueError(f"End effector body '{end_effector_body}' not found")
40
41 self._build_joint_mapping()
42
43 # IK parameters for numerical stability
44 self.damping, self.max_step_size = 0.01, 0.1

47 def _build_joint_mapping(self):
48 """
49 @brief Build mapping between joint names and their qpos indices
50
51 @note Uses joint naming patterns to identify SO100-specific joints
52 and defines control strategy based on joint functionality
53 """
54 self.joint_qpos_indices = {}
55
56 for i in range(self.model.njnt):
57 # Extract joint name from model
58 name_id = self.model.name_jntadr[i]
59 name_bytes = bytearray()
60 j = name_id
61 while j < len(self.model.names) and self.model.names[j] != 0:
62 name_bytes.append(self.model.names[j])
63 j += 1
64 joint_name = name_bytes.decode('utf-8') if name_bytes else f"joint_{i}"
65
66 # Map identifiable joints based on naming patterns
67 if 'Rotation' in joint_name:
68 self.joint_qpos_indices['rotation'] = i
69 elif 'Pitch' in joint_name and 'Wrist' not in joint_name:
70 self.joint_qpos_indices['pitch'] = i # Shoulder pitch
71 elif 'Elbow' in joint_name:
72 self.joint_qpos_indices['elbow'] = i
73 elif 'Wrist_Pitch' in joint_name:
74 self.joint_qpos_indices['wrist_pitch'] = i
75 elif 'Wrist_Roll' in joint_name:
76 self.joint_qpos_indices['wrist_roll'] = i
77
78 # Define control strategy: position joints vs manual joints
79 self.position_joints = ['pitch', 'elbow'] # Controlled by IK for translation
80 self.manual_joints = ['rotation', 'wrist_pitch', 'wrist_roll'] # Manual control
81
82 def solve_position_only(self, target_pos, fixed_rotation=None, fixed_wrist_roll=None,
83 fixed_wrist_pitch=None, max_iterations=50, tolerance=1e-3):
84 """
85 @brief Solve IK for position only while preserving manual joints
86
87 @param target_pos: Target position as 3D vector [x, y, z]
88 @param fixed_rotation: Fixed value for rotation joint (optional)
89 @param fixed_wrist_roll: Fixed value for wrist roll joint (optional)
90 @param fixed_wrist_pitch: Fixed value for wrist pitch joint (optional)
91 @param max_iterations: Maximum number of iterations
92 @param tolerance: Convergence tolerance for position error
93
94 @return: Tuple of (position_joint_values, success_flag)
95 - position_joint_values: Dictionary mapping joint names to values
96 - success_flag: Boolean indicating convergence success
97
98 @note Only solves for pitch and elbow joints while fixing other joints
99 to maintain manual control integrity
100 """
101 # Store original state for restoration
102 original_qpos = self.data.qpos.copy()
103 success = False
104
105 # Use current values if fixed values not provided
106 if fixed_rotation is None and 'rotation' in self.joint_qpos_indices:
107 fixed_rotation = self.data.qpos[self.joint_qpos_indices['rotation']]
108 if fixed_wrist_roll is None and 'wrist_roll' in self.joint_qpos_indices:
109 fixed_wrist_roll = self.data.qpos[self.joint_qpos_indices['wrist_roll']]
110 if fixed_wrist_pitch is None and 'wrist_pitch' in self.joint_qpos_indices:
111 fixed_wrist_pitch = self.data.qpos[self.joint_qpos_indices['wrist_pitch']]
112
113 for iteration in range(max_iterations):
114 # Apply fixed joint values to preserve manual control
115 if fixed_rotation is not None and 'rotation' in self.joint_qpos_indices:
116 self.data.qpos[self.joint_qpos_indices['rotation']] = fixed_rotation
117 if fixed_wrist_roll is not None and 'wrist_roll' in self.joint_qpos_indices:
118 self.data.qpos[self.joint_qpos_indices['wrist_roll']] = fixed_wrist_roll
119 if fixed_wrist_pitch is not None and 'wrist_pitch' in self.joint_qpos_indices:
120 self.data.qpos[self.joint_qpos_indices['wrist_pitch']] = fixed_wrist_pitch
121
122 # Update forward kinematics
123 mujoco.mj_forward(self.model, self.data)
124
125 # Calculate position error
126 current_pos = self.data.xpos[self.ee_body_id].copy()
127 pos_error = target_pos - current_pos
128 error_norm = np.linalg.norm(pos_error)
129
130 # Check convergence
131 if error_norm < tolerance:
132 success = True
133 break
134
135 # Compute position Jacobian only (3xN)
136 jac_pos = np.zeros((3, self.model.nv))
137 mujoco.mj_jacBody(self.model, self.data, jac_pos, None, self.ee_body_id)
138
139 # Extract controllable joints (position joints only)
140 controllable_joints = []
141 for joint_name in self.position_joints:
142 if joint_name in self.joint_qpos_indices:
143 joint_id = self.joint_qpos_indices[joint_name]
144 dof_adr = self.model.jnt_dofadr[joint_id]
145 controllable_joints.append(dof_adr)
146
147 # Exit if no controllable joints found
148 if not controllable_joints:
149 break
150
151 # Create reduced Jacobian for controllable joints
152 jacobian_reduced = jac_pos[:, controllable_joints]
153
154 # Damped least squares solution for position error
155 jacobian_t = jacobian_reduced.T
156 jjt = jacobian_reduced @ jacobian_t
157 damping_matrix = self.damping * np.eye(3)
158
159 try:
160 lambda_matrix = jjt + damping_matrix
161 jacobian_pinv = jacobian_t @ np.linalg.pinv(lambda_matrix)
162 delta_q = jacobian_pinv @ pos_error
163
164 # Limit step size for stability
165 step_norm = np.linalg.norm(delta_q)
166 if step_norm > self.max_step_size:
167 delta_q = delta_q * (self.max_step_size / step_norm)
168
169 # Apply updates to position joints with limits
170 for i, dof_adr in enumerate(controllable_joints):
171 for joint_name, qpos_idx in self.joint_qpos_indices.items():
172 if joint_name in self.position_joints:
173 joint_dof_adr = self.model.jnt_dofadr[qpos_idx]
174 if joint_dof_adr == dof_adr:
175 self.data.qpos[qpos_idx] += delta_q[i]
176 # Enforce joint limits
177 if self.model.jnt_limited[qpos_idx]:
178 qmin, qmax = self.model.jnt_range[qpos_idx]
179 self.data.qpos[qpos_idx] = np.clip(self.data.qpos[qpos_idx], qmin, qmax)
180 break
181 except np.linalg.LinAlgError:
182 # Handle singular Jacobian by breaking iteration
183 break
184
185 # Collect final position joint values
186 position_joint_values = {}
187 for joint_name in self.position_joints:
188 if joint_name in self.joint_qpos_indices:
189 position_joint_values[joint_name] = self.data.qpos[self.joint_qpos_indices[joint_name]]
190
191 # Restore original simulation state
192 self.data.qpos[:] = original_qpos
193 mujoco.mj_forward(self.model, self.data)
194
195 return position_joint_values, success
