86 def get_input_from_gamepad(self):
87 """
88 @brief Get modified input structure for SO100 pure translational control
89
90 @return: Dictionary containing all gamepad input values
91
92 @note Applies deadzone filtering and returns structured input data
93 """
94 DEADZONE_THRESHOLD = self.scales['deadzone_threshold']
95
96 def deadzone(value):
97 """Apply deadzone filtering to analog inputs"""
98 return 0.0 if abs(value) < DEADZONE_THRESHOLD else value
99
100 # Get raw inputs with deadzone filtering
101 left_y = deadzone(self.joystick.get_axis(1)) # Left stick vertical (pure Z)
102 right_y = deadzone(self.joystick.get_axis(3)) # Right stick vertical (horizontal in/out)
103 left_x = deadzone(-self.joystick.get_axis(0)) # Left stick horizontal (rotation)
104
105 # Button states for joint control
106 l1, r1 = self.joystick.get_button(4), self.joystick.get_button(5)
107 l2_raw, r2_raw = (self.joystick.get_axis(4) + 1) / 2, (self.joystick.get_axis(5) + 1) / 2
108 l2 = 0.0 if l2_raw < DEADZONE_THRESHOLD else l2_raw
109 r2 = 0.0 if r2_raw < DEADZONE_THRESHOLD else r2_raw
110
111 return {
112 'right_y': right_y, # Horizontal in/out relative to base
113 'left_y': left_y, # Pure vertical movement
114 'left_x': left_x, # Rotation joint control
115 'l1': l1, 'r1': r1, # Wrist pitch control
116 'l2': l2, 'r2': r2, # Wrist roll control
117 'start': self.joystick.get_button(7),
118 'a_button': self.joystick.get_button(0)
119 }

133 def process_translation_control(self, input_data):
134 """
135 @brief Process translational control using IK with conflict prevention
136
137 @param input_data: Dictionary containing gamepad input values
138
139 @note Skips translation when significant rotation input is detected
140 to prevent control conflicts between manual and IK control
141 """
142 # Conflict prevention: skip translation during active rotation
143 if abs(input_data['left_x']) > 0.2: # Rotation threshold
144 return # Ignore translation when rotating
145
146 # Process translation only if significant input detected
147 if abs(input_data['right_y']) > 0.01 or abs(input_data['left_y']) > 0.01:
148 ee_body = self.robot_config['end_effector_body']
149 current_pos, current_quat = self.sim.get_object_state(ee_body)
150
151 # Calculate target position based on pure translation
152 target_pos = self.calculate_pure_translation(
153 current_pos, input_data['right_y'], input_data['left_y'],
154 self.scales['translation'], self.sim.model.opt.timestep
155 )
156
157 # Preserve manual joint values for IK solution
158 current_rotation = self.sim.data.ctrl[self.joint_controller.joint_map['rotation']]
159 current_wrist_roll = self.sim.data.ctrl[self.joint_controller.joint_map['wrist_roll']]
160 current_wrist_pitch = self.sim.data.ctrl[self.joint_controller.joint_map['wrist_pitch']]
161
162 # Solve IK for position only with fixed manual joints
163 position_joint_values, success = self.ik_solver.solve_position_only(
164 target_pos, current_rotation, current_wrist_roll, current_wrist_pitch
165 )
166
167 if success:
168 # Apply solved joint positions to simulation
169 for joint_name, joint_value in position_joint_values.items():
170 if joint_name in self.joint_controller.joint_map:
171 self.sim.data.ctrl[self.joint_controller.joint_map[joint_name]] = joint_value

173 def calculate_pure_translation(self, current_pos, right_y, left_y, translation_scale, dt):
174 """
175 @brief Calculate target position using rotation joint as reference
176
177 @param current_pos: Current end-effector position
178 @param right_y: Right stick Y input (horizontal movement)
179 @param left_y: Left stick Y input (vertical movement)
180 @param translation_scale: Movement scaling factor
181 @param dt: Time step for integration
182
183 @return: Target position as 3D vector
184
185 @note Uses rotation joint position as reference for radial movement
186 """
187 target_pos = np.array(current_pos, dtype=np.float64)
188
189 # Get rotation joint position for radial movement reference
190 rotation_joint_pos = self.get_rotation_joint_position()
191
192 # Horizontal movement along radial direction from rotation joint
193 if abs(right_y) > 0.01:
194 horizontal_vec = np.array([current_pos[0] - rotation_joint_pos[0],
195 current_pos[1] - rotation_joint_pos[1]])
196 horizontal_dist = np.linalg.norm(horizontal_vec)
197
198 if horizontal_dist > 0.001:
199 # Move along radial direction
200 horizontal_dir = horizontal_vec / horizontal_dist
201 movement = right_y * translation_scale * dt * 50
202 target_pos[0] += horizontal_dir[0] * movement
203 target_pos[1] += horizontal_dir[1] * movement
204 else:
205 # Fallback for zero distance
206 target_pos[0] += right_y * translation_scale * dt * 50
207
208 # Pure vertical movement (independent of rotation)
209 if abs(left_y) > 0.01:
210 target_pos[2] += -left_y * translation_scale * dt * 50
211
212 return target_pos

214 def get_rotation_joint_position(self):
215 """
216 @brief Get world position of rotation joint for movement reference
217
218 @return: Position of rotation joint as 3D vector
219
220 @note Attempts multiple methods to find rotation joint position
221 """
222 # Try different methods to find rotation joint position
223 rotation_body_id = mujoco.mj_name2id(self.sim.model, mujoco.mjtObj.mjOBJ_BODY, "Rotation_Pitch")
224 if rotation_body_id != -1:
225 return self.sim.data.xpos[rotation_body_id].copy()
226
227 rotation_joint_id = mujoco.mj_name2id(self.sim.model, mujoco.mjtObj.mjOBJ_JOINT, "Rotation")
228 if rotation_joint_id != -1:
229 body_id = self.sim.model.jnt_bodyid[rotation_joint_id]
230 return self.sim.data.xpos[body_id].copy()
231
232 # Fallback to base position
233 base_body_id = mujoco.mj_name2id(self.sim.model, mujoco.mjtObj.mjOBJ_BODY, "Base")
234 if base_body_id != -1:
235 return self.sim.data.xpos[base_body_id].copy()
236
237 return np.array([0.0, 0.0, 0.0]) # Default origin

262 """
263 @brief Main teleoperation loop for SO100

268 if not self.initialize_systems():

283 # Process pygame events

285 if event.type == pygame.QUIT:

289 input_data = self.get_input_from_gamepad()

345 def _build_joint_map(self):
346 """
347 @brief Map joint names to their control indices
348
349 @return: Dictionary mapping joint names to actuator indices
350
351 @note Uses actuator naming patterns to identify SO100 joints
352 """
353 joint_map = {}
354 for i in range(self.sim.model.nu):
355 # Extract actuator name from model
356 name_id = self.sim.model.name_actuatoradr[i]
357 name_bytes = bytearray()
358 j = name_id
359 while j < len(self.sim.model.names) and self.sim.model.names[j] != 0:
360 name_bytes.append(self.sim.model.names[j])
361 j += 1
362 act_name = name_bytes.decode('utf-8') if name_bytes else f"actuator_{i}"
363
364 # Map to expected joints based on naming patterns
365 if 'Rotation' in act_name:
366 joint_map['rotation'] = i
367 elif 'Pitch' in act_name and 'Wrist' not in act_name:
368 joint_map['pitch'] = i
369 elif 'Elbow' in act_name:
370 joint_map['elbow'] = i
371 elif 'Wrist_Pitch' in act_name:
372 joint_map['wrist_pitch'] = i
373 elif 'Wrist_Roll' in act_name:
374 joint_map['wrist_roll'] = i
375 elif 'Jaw' in act_name:
376 joint_map['gripper'] = i
377
378 return joint_map

380 def control_rotation_joint(self, left_stick_x):
381 """
382 @brief Control base rotation joint
383
384 @param left_stick_x: Left stick horizontal input value
385
386 @note Applies movement scale and joint multiplier from configuration
387 """
388 if 'rotation' in self.joint_map:
389 scale = self.movement_scales['rotation'] * self.joint_multipliers['rotation']
390 current = self.sim.data.ctrl[self.joint_map['rotation']]
391 self.sim.data.ctrl[self.joint_map['rotation']] = current - left_stick_x * scale

393 def control_wrist_roll(self, l2, r2):
394 """
395 @brief Control wrist roll using triggers
396
397 @param l2: Left trigger value (0.0 to 1.0)
398 @param r2: Right trigger value (0.0 to 1.0)
399
400 @note Differential control: l2 decreases, r2 increases roll
401 """
402 if 'wrist_roll' in self.joint_map:
403 roll_scale = self.movement_scales['rotation'] * self.joint_multipliers['wrist_roll']
404 current = self.sim.data.ctrl[self.joint_map['wrist_roll']]
405 delta = (l2 - r2) * roll_scale
406 self.sim.data.ctrl[self.joint_map['wrist_roll']] = current + delta

408 def control_wrist_pitch(self, l1, r1):
409 """
410 @brief Control wrist pitch using shoulder buttons
411
412 @param l1: Left shoulder button state (0 or 1)
413 @param r1: Right shoulder button state (0 or 1)
414
415 @note Differential control: l1 decreases, r1 increases pitch
416 """
417 if 'wrist_pitch' in self.joint_map:
418 pitch_scale = self.movement_scales['tilt'] * self.joint_multipliers['wrist_pitch']
419 current = self.sim.data.ctrl[self.joint_map['wrist_pitch']]
420 delta = (l1 - r1) * pitch_scale
421 self.sim.data.ctrl[self.joint_map['wrist_pitch']] = current + delta
