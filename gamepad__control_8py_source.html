51 def initialize_teleop_system(self):
52 """
53 @brief Initialize appropriate teleoperation system based on selected robot
54
55 @return: True if initialization successful, False otherwise
56
57 @throws ImportError: If required teleoperation module cannot be imported
58 @throws Exception: For general initialization failures
59
60 @note Uses SO100-specific system for SO100, generic for Panda/UR5

62 if not self.selected_robot:
63 print("Error: No robot selected. Please call select_robot() first.")
64 return False
65
66 try:
67 if self.selected_robot == 'so100':
68 # Use SO100-specific teleoperation
69 from so100teleoperation import SO100Teleoperation
70 self.teleop_system = SO100Teleoperation()
71 print("Initialized SO100 teleoperation system")
72 else:
73 # Use generic teleoperation for Panda/UR5
74 from genericteleoperation import GenericTeleoperation
75 self.teleop_system = GenericTeleoperation(robot_name=self.selected_robot)
76 print(f"Initialized Generic teleoperation system for {self.selected_robot}")
77
78 return True
79
80 except ImportError as e:
81 print(f"Error importing teleoperation module: {e}")
82 return False
83 except Exception as e:
84 print(f"Error initializing teleoperation system: {e}")
85 return False

88 """
89 @brief Execute the selected teleoperation system
