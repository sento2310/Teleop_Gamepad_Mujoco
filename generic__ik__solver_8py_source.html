10from scipy.spatial.transform import Rotation as R

12
13class GenericVelocityIKSolver:
14 """
15 @brief Inverse kinematics solver using Jacobian pseudo-inverse method
16 @details Implements iterative damped least squares for end-effector pose control
17 with numerical stability and joint limit constraints.
18 """
19
20 def __init__(self, model, data, end_effector_body, arm_joint_count=None):
21 """
22 @brief Initialize IK solver for specified end effector
23
24 @param model: MuJoCo model object containing robot kinematics
25 @param data: MuJoCo data object for current simulation state
26 @param end_effector_body: Name of end effector body for IK targeting
27 @param arm_joint_count: Number of arm joints to use for IK solution
28
29 @throws ValueError: If end_effector_body not found in model
30
31 @note Automatically detects revolute joints for IK control
32 """
33 self.model, self.data = model, data
34
35 # Get end-effector body ID
36 self.ee_body_id = mujoco.mj_name2id(model, mujoco.mjtObj.mjOBJ_BODY, end_effector_body)
37 if self.ee_body_id == -1:
38 raise ValueError(f"End effector body '{end_effector_body}' not found")
39
40 # Find revolute joints for IK control
41 all_joint_indices = [i for i in range(model.njnt)
42 if model.jnt_type[i] == mujoco.mjtJoint.mjJNT_HINGE]
43
44 self.arm_joint_count = min(arm_joint_count, len(all_joint_indices)) if arm_joint_count else len(all_joint_indices)
45 self.joint_indices = all_joint_indices[:self.arm_joint_count]
46 self.n_joints = len(self.joint_indices)
47
48 # IK parameters for numerical stability
49 self.damping, self.max_step_size = 0.01, 0.1
50
51 def solve(self, target_pos, target_quat, max_iterations=50, tolerance=1e-3):
52 """
53 @brief Solve inverse kinematics for target end-effector pose
54
55 @param target_pos: Target position as 3D vector [x, y, z]
56 @param target_quat: Target orientation as quaternion [w, x, y, z]
57 @param max_iterations: Maximum number of gradient descent iterations
58 @param tolerance: Convergence tolerance for pose error norm
59
60 @return: Tuple of (joint_positions, success_flag)
61 - joint_positions: Array of solved joint angles
62 - success_flag: Boolean indicating convergence success
63
64 @throws LinAlgError: If Jacobian matrix is singular during inversion
65
66 @note Uses damped least squares: $\Delta q = J^T (JJ^T + \lambda I)^{-1} e$
67 where $e$ is the 6D pose error vector and $\lambda$ is damping factor
68 """
69 if self.n_joints == 0:
70 return np.array([]), False
71
72 # Store original state for restoration
73 original_qpos = self.data.qpos.copy()
74 success = False
75
76 for iteration in range(max_iterations):
77 mujoco.mj_forward(self.model, self.data)
78
79 # Current pose and errors
80 current_pos = self.data.xpos[self.ee_body_id].copy()
81 current_quat = self.data.xquat[self.ee_body_id].copy()
82
83 # Position error
84 pos_error = target_pos - current_pos
85
86 # Orientation error as rotation vector
87 current_rot = R.from_quat([current_quat[1], current_quat[2], current_quat[3], current_quat[0]])
88 target_rot = R.from_quat([target_quat[1], target_quat[2], target_quat[3], target_quat[0]])
89 orient_error = (target_rot * current_rot.inv()).as_rotvec()
90
91 # Combined error vector
92 error = np.concatenate([pos_error, orient_error])
93 error_norm = np.linalg.norm(error)
94
95 # Check convergence
96 if error_norm < tolerance:
97 success = True
98 break
99
100 # Compute Jacobian matrices
101 jac_pos, jac_rot = np.zeros((3, self.model.nv)), np.zeros((3, self.model.nv))
102 mujoco.mj_jacBody(self.model, self.data, jac_pos, jac_rot, self.ee_body_id)
103 jacobian = np.vstack([jac_pos, jac_rot])
104
105 # Extract controllable joints subspace
106 jacobian_reduced = np.zeros((6, self.n_joints))
107 for i, joint_id in enumerate(self.joint_indices):
108 dof_adr = self.model.jnt_dofadr[joint_id]
109 jacobian_reduced[:, i] = jacobian[:, dof_adr]
110
111 # Damped least squares solution
112 jacobian_t = jacobian_reduced.T
113 jjt = jacobian_reduced @ jacobian_t
114 damping_matrix = self.damping * np.eye(6)
115
116 try:
117 lambda_matrix = jjt + damping_matrix
118 jacobian_pinv = jacobian_t @ np.linalg.pinv(lambda_matrix)
119 delta_q = jacobian_pinv @ error
120
121 # Limit step size for stability
122 step_norm = np.linalg.norm(delta_q)
123 if step_norm > self.max_step_size:
124 delta_q = delta_q * (self.max_step_size / step_norm)
125
126 # Apply joint updates with limits
127 for i, joint_id in enumerate(self.joint_indices):
128 self.data.qpos[joint_id] += delta_q[i]
129 if self.model.jnt_limited[joint_id]:
130 qmin, qmax = self.model.jnt_range[joint_id]
131 self.data.qpos[joint_id] = np.clip(self.data.qpos[joint_id], qmin, qmax)
132
133 except np.linalg.LinAlgError:
134 # Fallback for singularities with increased damping
135 damping_matrix = 0.1 * np.eye(6)
136 lambda_matrix = jjt + damping_matrix
137 jacobian_pinv = jacobian_t @ np.linalg.pinv(lambda_matrix)
138 delta_q = jacobian_pinv @ error * 0.1
139
140 for i, joint_id in enumerate(self.joint_indices):
141 self.data.qpos[joint_id] += delta_q[i]
142
143 # Get final joint positions and restore simulation state
144 final_joint_pos = self.data.qpos[self.joint_indices].copy()
145 self.data.qpos[:] = original_qpos
146 mujoco.mj_forward(self.model, self.data)
147
148 return final_joint_pos, success
