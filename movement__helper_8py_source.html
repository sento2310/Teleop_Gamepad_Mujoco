41 def _find_gripper_controls(self):
42 """
43 @brief Find control indices for gripper actuators
44
45 @return: List of actuator indices controlling the gripper
46
47 @note Uses robot-specific naming conventions to identify gripper actuators
48 """
49 gripper_indices = []
50
51 for i in range(self.sim.model.nu):
52 # Extract actuator name from model
53 name_id = self.sim.model.name_actuatoradr[i]
54 name_bytes = bytearray()
55 j = name_id
56 while j < len(self.sim.model.names) and self.sim.model.names[j] != 0:
57 name_bytes.append(self.sim.model.names[j])
58 j += 1
59 act_name = name_bytes.decode('utf-8') if name_bytes else f"actuator_{i}"
60
61 # Robot-specific gripper detection
62 if self.robot_name == 'so100':
63 if 'Jaw' in act_name:
64 gripper_indices.append(i)
65 else:
66 # Generic gripper detection for Panda/UR5
67 if any(keyword in act_name.lower() for keyword in ['finger', 'gripper', 'hand']):
68 gripper_indices.append(i)
69
70 # Default to last actuator if none found
71 if not gripper_indices:
72 gripper_indices = [self.sim.model.nu - 1]
73
74 return gripper_indices

86 def integrate_twist(self, twist_command):
87 """
88 @brief Integrate twist command to update target pose
89
90 @param twist_command: 6D twist [vx, vy, vz, wx, wy, wz] in end-effector frame
91 @return: Tuple of (new_position, new_orientation)
92
93 @note Applies speed limits and transforms velocities to world frame
94 """
95 if self.target_pos is None or self.target_quat is None:
96 raise ValueError("Initial pose not set")
97
98 # Extract and limit linear/angular velocities
99 linear_vel = np.array(twist_command[:3])
100 angular_vel = np.array(twist_command[3:])
101
102 # Apply speed limits
103 linear_speed = np.linalg.norm(linear_vel)
104 if linear_speed > self.max_linear_speed:
105 linear_vel = linear_vel * (self.max_linear_speed / linear_speed)
106
107 angular_speed = np.linalg.norm(angular_vel)
108 if angular_speed > self.max_angular_speed:
109 angular_vel = angular_vel * (self.max_angular_speed / angular_speed)
110
111 # Convert current orientation to rotation object
112 current_rot = R.from_quat([self.target_quat[1], self.target_quat[2],
113 self.target_quat[3], self.target_quat[0]])
114
115 # Transform velocities to world frame
116 linear_vel_world = current_rot.apply(linear_vel)
117 new_pos = self.target_pos + linear_vel_world * self.dt
118
119 # Update orientation using exponential map
120 if np.linalg.norm(angular_vel) > 1e-6:
121 angular_vel_world = current_rot.apply(angular_vel)
122 delta_angle = angular_vel_world * self.dt
123 delta_rotation = R.from_rotvec(delta_angle)
124 new_rot = delta_rotation * current_rot
125 new_quat_scipy = new_rot.as_quat() # [x, y, z, w] format
126 new_quat = np.array([new_quat_scipy[3], new_quat_scipy[0],
127 new_quat_scipy[1], new_quat_scipy[2]]) # Convert to [w, x, y, z]
128 else:
129 new_quat = self.target_quat.copy()
130
131 # Normalize quaternion and update target
132 new_quat /= np.linalg.norm(new_quat)
133 self.target_pos, self.target_quat = new_pos, new_quat
134
135 return new_pos, new_quat

145 def move_gripper(self, gripper_pos, speed):
146 """
147 @brief Set target gripper position and movement speed
148
149 @param gripper_pos: Target position (0.0=closed, 1.0=open)
150 @param speed: Movement speed (0-1 scale)
151
152 @note Converts normalized position to actuator control range
153 """
154 max_speed = 255.0 # Maximum speed in actuator units per second
155 self.gripper_target = np.clip(gripper_pos * 255, 0, 255)
156 self.gripper_speed = np.clip(speed * max_speed, 0, max_speed)

158 def update_gripper(self):
159 """
160 @brief Update gripper position towards target with speed control
161
162 @note Moves gripper incrementally towards target position each call
163 """
164 if not self.gripper_control_indices:
165 return
166
167 for control_index in self.gripper_control_indices:
168 if control_index >= len(self.sim.data.ctrl):
169 continue
170
171 current_pos = self.sim.data.ctrl[control_index]
172 delta = self.gripper_target - current_pos
173
174 # Skip if already at target
175 if abs(delta) < 0.1:
176 continue
177
178 # Calculate step size based on speed and timestep
179 step_size = self.gripper_speed * self.sim.model.opt.timestep
180
181 # Move towards target with acceleration limiting
182 if abs(delta) <= step_size:
183 new_pos = self.gripper_target # Snap to target if close
184 else:
185 new_pos = current_pos + np.sign(delta) * step_size
186
187 self.sim.data.ctrl[control_index] = new_pos
